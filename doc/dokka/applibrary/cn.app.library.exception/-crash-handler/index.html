class CrashHandler : Thread.UncaughtExceptionHandler

捕获异常 使用方法： 1、在 Application 类onCreate() 方法中初始化 CrashHandler import android.app.Application

class MyApplication : Application() {

override fun onCreate() {
    super.onCreate()

    // 指定崩溃日志的保存目录
    val crashLogDirectory = "CrashLogs"

    // 获取 CrashHandler 实例并初始化
    val crashHandler = CrashHandler.getInstance(crashLogDirectory)
    crashHandler.init(this)
}
